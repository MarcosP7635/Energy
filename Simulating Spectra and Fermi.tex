\begin{equation}

\begin{equation}
\begin{split}
\int_0^{T_{e \ max}} N(T_e)dT_e\\
\int_0^{T_{e \ max}} \frac{C}{c^5}(Q - T_e)^2(T_e + m_ec^2)\sqrt{T_e^2 + 2T_em_ec^2} dT_e = 1\\
